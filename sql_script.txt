CREATE TABLE PULS
(TreningsoktsId CHAR(5),
Id CHAR(5),
Sekunder_etter_start SMALLINT() UNSIGNED,
Puls TINYINT() UNSIGNED,
CONSTRAINT Puls_PK PRIMARY KEY(TreningsoktsId, Id),
CONSTRAINT Puls_FK REFERENCES TRENINGSOKT(Id)
);

CREATE TABLE TRENINGSOKT
(Id CHAR(5),
Dato_tid DATETIME(),
Personlig_form TEXT,
Prestasjon TEXT,
Notat TEXT,
CONSTRAINT Treningsokt_PK PRIMARY KEY(Id)
);

CREATE TABLE OVELSE
(Id CHAR(5),
Navn VARCHAR(30),
Beskrivelse TEXT,
Type BOOLEAN,
CONSTRAINT Ovelse_PK PRIMARY KEY(Id)
);

CREATE TABLE GRUPPE
(Id CHAR(5),
Navn VARCHAR(30),
Beskrivelse TEXT,
CONSTRAINT Gruppe_PK PRIMARY KEY(Id)
);
