CREATE TABLE PULS
(Id CHAR(5),
Sekunder_etter_start SMALLINT UNSIGNED,
Puls TINYINT UNSIGNED,
CONSTRAINT Puls_PK PRIMARY KEY(Id, Sekunder_etter_start),
CONSTRAINT Puls_FK FOREIGN KEY(Id) REFERENCES TRENINGSOKT(Id)
);

CREATE TABLE TRENINGSOKT
(Id CHAR(5),
Dato_tid DATETIME,
Personlig_form TEXT,
Prestasjon TEXT,
Notat TEXT,
CONSTRAINT Treningsokt_PK PRIMARY KEY(Id)
);

CREATE TABLE OVELSE
(Id CHAR(5),
Navn VARCHAR(30),
Beskrivelse TEXT,
Type BOOLEAN,
CONSTRAINT Ovelse_PK PRIMARY KEY(Id)
);

CREATE TABLE GRUPPE
(Id CHAR(5),
Navn VARCHAR(30),
Beskrivelse TEXT,
CONSTRAINT Gruppe_PK PRIMARY KEY(Id)
);

CREATE TABLE HAR_OVELSE
(FOREIGN KEY (Ovelse_id) REFERENCES OVELSE(Id),
FOREIGN KEY (Treningsprogrammal_navn) REFERENCES TRENINGSPROGRAMMAL(navn)
);

CREATE TABLE DEL_AV_GRUPPE
(FOREIGN KEY (Gruppe_id) REFERENCES GRUPPE(id),
FOREIGN KEY (Ovelse_id) REFERENCES OVELSE(Id)
);

CREATE TABLE STYRKERESULTAT
(Id int NOT NULL,
Belastning DECIMAL(3,1) NOT NULL,
Repetisjoner INT(3) NOT NULL,
Sett INT(2) NOT NULL,
(PRIMARY KEY (Id),
FOREIGN KEY (Ovelse_id) REFERENCES Ovelse(Id),
FOREIGN KEY (Treningsokt_id) REFERENCES Treningsokt(Id),
);

CREATE TABLE KONDISJONSRESULTAT
(Id INT NOT NULL,
Lengde INT NOT NULL,
Tid INT NOT NULL,
PRIMARY KEY (Id)
FOREIGN KEY (Ovelse_id) REFERENCES Ovelse(Id),
FOREIGN KEY (Treningsokt_id) REFERENCES Treningsokt(Id),
);

CREATE TABLE STYRKEMAL
(Id INT NOT NULL,
Belastning DECIMAL(3,1) NOT NULL,
Repetisjoner INT(3) NOT NULL,
Sett INT(2) NOT NULL,
PRIMARY KEY (Id),
FOREIGN KEY (Ovelse_id) REFERENCES Ovelse(Id)
);

CREATE TABLE KONDISJONSMAL
(Id INT NOT NULL,
Lengde INT NOT NULL,
Tid INT NOT NULL,
PRIMARY KEY (Id),
FOREIGN KEY (Ovelse_id) REFERENCES Ovelse(Id)
);
